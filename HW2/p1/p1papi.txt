#!/bin/sh
##SBATCH --partition=general-compute
#SBATCH --time=00:30:00
#SBATCH --nodes=1
#SBATCH --ntasks-per-node=8
##SBATCH --mem=24000
# Memory per node specification is in MB. It is optional.
# The default limit is 3GB per core.
#SBATCH --job-name="p1papi"
#SBATCH --output=p1papi.out
#SBATCH --mail-user=saphale@buffalo.edu
#SBATCH --mail-type=ALL
#SBATCH --requeue
#Specifies that the job will be requeued after a node failure.
#The default is that the job will not be requeued.

echo "SLURM_JOBID="$SLURM_JOBID
echo "SLURM_JOB_NODELIST"=$SLURM_JOB_NODELIST
echo "SLURM_NNODES"=$SLURM_NNODES
echo "SLURMTMPDIR="$SLURMTMPDIR

cd $SLURM_SUBMIT_DIR
echo "working directory = "$SLURM_SUBMIT_DIR

module load gcc
module load intel/16.0
module load papi
module list
ulimit -s unlimited
#

echo "p1papi"
export I_MPI_PMI_LIBRARY=/usr/lib64/libpmi.so
/var/lib/pcp/pmdas/perfevent/perfalloc &

icpc -Wall -I$PAPI/include -o Run3 p1cpapi.cpp timer.C -L$PAPI/lib -lpapi

icpc -Wall -I$PAPI/include -o Run4 p1rpapi.cpp timer.C -L$PAPI/lib -lpapi

srun -n 1 ./Run3

srun -n 1 ./Run4


echo "All compiled"


#
echo "All Done"
